// JavaScript
fetch("/generate-resume", {
  method: "POST",
  headers: { "Content-Type": "application/json" },
  body: JSON.stringify(window.resumeData),
})
  .then(r => r.json()) // or .text() if HTML
  .then(data => {
    console.log("Backend result:", data);
  });
